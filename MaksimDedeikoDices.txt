void sum(int a, int b)
{
	int s1 = 0;
	int s2 = 0;
	for (int i = a; i <= a; i++)
	{
		s1 = s1 + i;
	}
	for (int i = b; i <= b; i++)
	{
		s2 = s2 + i;
	}
	cout << "Ваша сумма значений: " << s1 << endl;
	cout << "Значение компьютера: " << s2 << endl;
}
