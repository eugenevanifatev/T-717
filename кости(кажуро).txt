bool ktopervey()
{
	int step = rand() % 2;

	if (step == 1)
	{
		cout << "Первый ход: Игрок" << endl << endl;

	else
	{
		cout << "Первый ход: Компьютер" << endl << endl;
	}

}

void showCurrentResult(int &score, int EndResultPlayer, int EndResultComputer)
{
	if (score>1)
	{
		if (EndResultPlayer >EndResultComputer)
		{
			cout << "Выйграл игрок" << endl << endl;
			score = 0;
		}
		else if (EndResultPlayer == EndResultComputer)
		{
			cout << "Ничья" << endl << endl;
			score = 0;
		}
		else
		{
			cout << "Выйграл компьютер" << endl << endl;
			score = 0;
		}
	}
}
